// -------------------------------------------------------
//
//  CLASS ArmorsDatas
//
// -------------------------------------------------------

/** @class
*   All the armors datas.
*   @property {SystemArmor[]} list List of all the armors of the game.
*/
function ArmorsDatas(){
    this.read();
}

ArmorsDatas.prototype = {

    read: function(){
        Wanok.openFile(this, Wanok.FILE_ARMORS, true, function(res){
            var json = JSON.parse(res)["armors"];
            var i, l = json.length;
            this.list = new Array(l+1);

            for (i = 0; i < l; i++){
                var jsonArmor = json[i];
                var id = jsonArmor["id"];
                var armor = new SystemArmor();
                armor.readJSON(jsonArmor);
                this.list[id] = armor;
            }
        });
    }
}

// -------------------------------------------------------
//
//  CLASS SystemArmor
//
//  An armor of the game.
//
//  @name     -> The name of the armor
//  @idType   -> The kind of armor (id)
//
// -------------------------------------------------------

function SystemArmor(){

}

SystemArmor.prototype = {

    getType: function(){
        return $gameDatas.battleSystemDatas.armorsKind[this.idType];
    },

    // -------------------------------------------------------

    readJSON: function(json){
        this.name = json.names[1];
        this.idType = json.k;
    }
}
